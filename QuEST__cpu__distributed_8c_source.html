35Complex statevec_calcInnerProduct(Qureg bra, Qureg ket) {
36
37 Complex localInnerProd = statevec_calcInnerProductLocal(bra, ket);
38 if (bra.numChunks == 1)
39 return localInnerProd;
40
41 qreal localReal = localInnerProd.real;
42 qreal localImag = localInnerProd.imag;
43 qreal globalReal, globalImag;
44 MPI_Allreduce(&localReal, &globalReal, 1, MPI_QuEST_REAL, MPI_SUM, MPI_COMM_WORLD);
45 MPI_Allreduce(&localImag, &globalImag, 1, MPI_QuEST_REAL, MPI_SUM, MPI_COMM_WORLD);
46
47 Complex globalInnerProd;
48 globalInnerProd.real = globalReal;
49 globalInnerProd.imag = globalImag;
50 return globalInnerProd;
51}

53qreal densmatr_calcTotalProb(Qureg qureg) {
54
55 // computes the trace by summing every element ("diag") with global index (2^n + 1)i for i in [0, 2^n-1]
56
57 long long int firstGlobalInd = qureg.chunkId * qureg.numAmpsPerChunk;
58 long long int diagSpacing = 1LL + (1LL << qureg.numQubitsRepresented);
59 long long int distFromPrevDiag = (firstGlobalInd % diagSpacing);
60 long long int distToNextDiag = (distFromPrevDiag > 0)? diagSpacing - distFromPrevDiag : 0;
61
62 // Kahan parameters
63 qreal rankTotal = 0;
64 qreal y, t, c;
65 c = 0;
66
67 long long int index = distToNextDiag;
68
69 while (index < qureg.numAmpsPerChunk) {
70
71 // Kahan summation - brackets are important
72 y = qureg.stateVec.real[index] - c;
73 t = rankTotal + y;
74 c = ( t - rankTotal ) - y;
75 rankTotal = t;
76
77 index += diagSpacing;
78 }
79
80 // combine each node's sum of diagonals
81 qreal globalTotal;
82 if (qureg.numChunks > 1)
83 MPI_Allreduce(&rankTotal, &globalTotal, 1, MPI_QuEST_REAL, MPI_SUM, MPI_COMM_WORLD);
84 else
85 globalTotal = rankTotal;
86
87 return globalTotal;
88}

90qreal statevec_calcTotalProb(Qureg qureg){
91 // Implemented using Kahan summation for greater accuracy at a slight floating
92 // point operation overhead. For more details see https://en.wikipedia.org/wiki/Kahan_summation_algorithm
93 qreal pTotal=0;
94 qreal y, t, c;
95 qreal allRankTotals=0;
96 long long int index;
97 long long int numAmpsPerRank = qureg.numAmpsPerChunk;
98 c = 0.0;
99 for (index=0; index<numAmpsPerRank; index++){
100 // Perform pTotal+=qureg.stateVec.real[index]*qureg.stateVec.real[index]; by Kahan
101 y = qureg.stateVec.real[index]*qureg.stateVec.real[index] - c;
102 t = pTotal + y;
103 // Don't change the bracketing on the following line
104 c = ( t - pTotal ) - y;
105 pTotal = t;
106 // Perform pTotal+=qureg.stateVec.imag[index]*qureg.stateVec.imag[index]; by Kahan
107 y = qureg.stateVec.imag[index]*qureg.stateVec.imag[index] - c;
108 t = pTotal + y;
109 // Don't change the bracketing on the following line
110 c = ( t - pTotal ) - y;
111 pTotal = t;
112 }
113 if (qureg.numChunks>1)
114 MPI_Allreduce(&pTotal, &allRankTotals, 1, MPI_QuEST_REAL, MPI_SUM, MPI_COMM_WORLD);
115 else
116 allRankTotals=pTotal;
117
118 return allRankTotals;
119}

132
133 QuESTEnv env;
134
135 // init MPI environment
136 int rank, numRanks, initialized;
137 MPI_Initialized(&initialized);
138 if (!initialized){
139 MPI_Init(NULL, NULL);
140 MPI_Comm_size(MPI_COMM_WORLD, &numRanks);
141 MPI_Comm_rank(MPI_COMM_WORLD, &rank);
142
143 env.rank=rank;
144 env.numRanks=numRanks;
145
146 } else {
147
148 printf("ERROR: Trying to initialize QuESTEnv multiple times. Ignoring...\n");
149
150 // ensure env is initialised anyway, so the compiler is happy
151 MPI_Comm_size(MPI_COMM_WORLD, &numRanks);
152 MPI_Comm_rank(MPI_COMM_WORLD, &rank);
153 env.rank=rank;
154 env.numRanks=numRanks;
155 }
156
157 validateNumRanks(env.numRanks, __func__);
158
159 env.seeds = NULL;
160 env.numSeeds = 0;
161 seedQuESTDefault(&env);
162
163 return env;
164}

387void copyVecIntoMatrixPairState(Qureg matr, Qureg vec) {
388
389 // Remember that for every amplitude that `vec` stores on the node,
390 // `matr` stores an entire column. Ergo there are always an integer
391 // number (in fact, a power of 2) number of `matr`s columns on each node.
392 // Since the total size of `vec` (between all nodes) is one column
393 // and each node stores (possibly) multiple columns (vec.numAmpsPerChunk as many),
394 // `vec` can be fit entirely inside a single node's matr.pairStateVec (with excess!)
395
396 // copy this node's vec segment into this node's matr pairState (in the right spot)
397 long long int numLocalAmps = vec.numAmpsPerChunk;
398 long long int myOffset = vec.chunkId * numLocalAmps;
399 memcpy(&matr.pairStateVec.real[myOffset], vec.stateVec.real, numLocalAmps * sizeof(qreal));
400 memcpy(&matr.pairStateVec.imag[myOffset], vec.stateVec.imag, numLocalAmps * sizeof(qreal));
401
402 // we now want to share this node's vec segment with other node, so that
403 // vec is cloned in every node's matr.pairStateVec
404
405 // work out how many messages needed to send vec chunks (2GB limit)
406 long long int maxMsgSize = MPI_MAX_AMPS_IN_MSG;
407 if (numLocalAmps < maxMsgSize)
408 maxMsgSize = numLocalAmps;
409 // safely assume MPI_MAX... = 2^n, so division always exact:
410 int numMsgs = numLocalAmps / maxMsgSize;
411
412 // every node gets a turn at being the broadcaster
413 for (int broadcaster=0; broadcaster < vec.numChunks; broadcaster++) {
414
415 long long int otherOffset = broadcaster * numLocalAmps;
416
417 // every node sends a slice of qureg's pairState to every other
418 for (int i=0; i< numMsgs; i++) {
419
420 // by sending that slice in further slices (due to bandwidth limit)
421 MPI_Bcast(
422 &matr.pairStateVec.real[otherOffset + i*maxMsgSize],
423 maxMsgSize, MPI_QuEST_REAL, broadcaster, MPI_COMM_WORLD);
424 MPI_Bcast(
425 &matr.pairStateVec.imag[otherOffset + i*maxMsgSize],
426 maxMsgSize, MPI_QuEST_REAL, broadcaster, MPI_COMM_WORLD);
427 }
428 }
429}

496 // MPI send/receive vars
497 int TAG=100;
498 MPI_Status status;
499
500 // Multiple messages are required as MPI uses int rather than long long int for count
501 // For openmpi, messages are further restricted to 2GB in size -- do this for all cases

559void compressPairVectorForSingleQubitDepolarise(Qureg qureg, int targetQubit){
560 long long int sizeInnerBlock, sizeInnerHalfBlock;
561 long long int sizeOuterColumn, sizeOuterHalfColumn;
562 long long int thisInnerBlock, // current block
563 thisOuterColumn, // current column in density matrix
564 thisIndex, // current index in (density matrix representation) state vector
565 thisIndexInOuterColumn,
566 thisIndexInInnerBlock;
567
568 int outerBit;
569
570 long long int thisTask;
571 long long int numTasks=qureg.numAmpsPerChunk>>1;
572
573 // set dimensions
574 sizeInnerHalfBlock = 1LL << targetQubit;
575 sizeInnerBlock = 2LL * sizeInnerHalfBlock;
576 sizeOuterHalfColumn = 1LL << qureg.numQubitsRepresented;
577 sizeOuterColumn = 2LL * sizeOuterHalfColumn;
578
579# ifdef _OPENMP
580# pragma omp parallel \
581 default (none) \
582 shared (sizeInnerBlock,sizeInnerHalfBlock,sizeOuterColumn,sizeOuterHalfColumn, \
583 qureg,numTasks,targetQubit) \
584 private (thisTask,thisInnerBlock,thisOuterColumn,thisIndex,thisIndexInOuterColumn, \
585 thisIndexInInnerBlock,outerBit)
586# endif
587 {
588# ifdef _OPENMP
589# pragma omp for schedule (static)
590# endif
591 // thisTask iterates over half the elements in this process' chunk of the density matrix
592 // treat this as iterating over all columns, then iterating over half the values
593 // within one column.
594 // If this function has been called, this process' chunk contains half an
595 // outer block or less
596 for (thisTask=0; thisTask<numTasks; thisTask++) {
597 // we want to process all columns in the density matrix,
598 // updating the values for half of each column (one half of each inner block)
599 thisOuterColumn = thisTask / sizeOuterHalfColumn;
600 thisIndexInOuterColumn = thisTask&(sizeOuterHalfColumn-1); // thisTask % sizeOuterHalfColumn
601 thisInnerBlock = thisIndexInOuterColumn/sizeInnerHalfBlock;
602 // get index in state vector corresponding to upper inner block
603 thisIndexInInnerBlock = thisTask&(sizeInnerHalfBlock-1); // thisTask % sizeInnerHalfBlock
604 thisIndex = thisOuterColumn*sizeOuterColumn + thisInnerBlock*sizeInnerBlock
605 + thisIndexInInnerBlock;
606 // check if we are in the upper or lower half of an outer block
607 outerBit = extractBit(targetQubit, (thisIndex+qureg.numAmpsPerChunk*qureg.chunkId)>>qureg.numQubitsRepresented);
608 // if we are in the lower half of an outer block, shift to be in the lower half
609 // of the inner block as well (we want to dephase |0><0| and |1><1| only)
610 thisIndex += outerBit*(sizeInnerHalfBlock);
611
612 // NOTE: at this point thisIndex should be the index of the element we want to
613 // dephase in the chunk of the state vector on this process, in the
614 // density matrix representation.
615 // thisTask is the index of the pair element in pairStateVec
616 // we will populate the second half of pairStateVec with this process'
617 // data to send
618
619 qureg.pairStateVec.real[thisTask+numTasks] = qureg.stateVec.real[thisIndex];
620 qureg.pairStateVec.imag[thisTask+numTasks] = qureg.stateVec.imag[thisIndex];
621
622 }
623 }
624}

662 {
663# ifdef _OPENMP
664# pragma omp for schedule (static)
665# endif
666 // thisTask iterates over half the elements in this process' chunk of the density matrix
667 // treat this as iterating over all columns, then iterating over half the values
668 // within one column.
669 // If this function has been called, this process' chunk contains half an
670 // outer block or less
671 for (thisTask=0; thisTask<numTasks; thisTask++) {
672 // we want to process all columns in the density matrix,
673 // updating the values for half of each column (one half of each inner block)
674 thisOuterColumn = thisTask / sizeOuterQuarterColumn;
675 // thisTask % sizeOuterQuarterColumn
676 thisIndexInOuterColumn = thisTask&(sizeOuterQuarterColumn-1);
677 thisInnerBlockQ2 = thisIndexInOuterColumn / sizeInnerQuarterBlockQ2;
678 // thisTask % sizeInnerQuarterBlockQ2;
679 thisIndexInInnerBlockQ2 = thisTask&(sizeInnerQuarterBlockQ2-1);
680 thisInnerBlockQ1InInnerBlockQ2 = thisIndexInInnerBlockQ2 / sizeInnerHalfBlockQ1;
681 // thisTask % sizeInnerHalfBlockQ1;
682 thisIndexInInnerBlockQ1 = thisTask&(sizeInnerHalfBlockQ1-1);
683
684 // get index in state vector corresponding to upper inner block
685 thisIndex = thisOuterColumn*sizeOuterColumn + thisInnerBlockQ2*sizeInnerBlockQ2
686 + thisInnerBlockQ1InInnerBlockQ2*sizeInnerBlockQ1 + thisIndexInInnerBlockQ1;
687
688 // check if we are in the upper or lower half of an outer block for Q1
689 outerBitQ1 = extractBit(targetQubit, (thisIndex+qureg.numAmpsPerChunk*qureg.chunkId)>>qureg.numQubitsRepresented);
690 // if we are in the lower half of an outer block, shift to be in the lower half
691 // of the inner block as well (we want to dephase |0><0| and |1><1| only)
692 thisIndex += outerBitQ1*(sizeInnerHalfBlockQ1);
693
694 // check if we are in the upper or lower half of an outer block for Q2
695 outerBitQ2 = extractBit(qubit2, (thisIndex+qureg.numAmpsPerChunk*qureg.chunkId)>>qureg.numQubitsRepresented);
696 // if we are in the lower half of an outer block, shift to be in the lower half
697 // of the inner block as well (we want to dephase |0><0| and |1><1| only)
698 thisIndex += outerBitQ2*(sizeInnerQuarterBlockQ2<<1);
699
700 // NOTE: at this point thisIndex should be the index of the element we want to
701 // dephase in the chunk of the state vector on this process, in the
702 // density matrix representation.
703 // thisTask is the index of the pair element in pairStateVec
704
705 // state[thisIndex] = (1-depolLevel)*state[thisIndex] + depolLevel*(state[thisIndex]
706 // + pair[thisTask])/2
707 qureg.pairStateVec.real[thisTask+numTasks*2] = qureg.stateVec.real[thisIndex];
708 qureg.pairStateVec.imag[thisTask+numTasks*2] = qureg.stateVec.imag[thisIndex];
709 }
710 }

1099void statevec_multiControlledMultiQubitNot(Qureg qureg, int ctrlMask, int targMask)
1100{
1101 /* operation is the same regardless of control and target ordering, hence
1102 * we accept only bitMasks (for convenience of caller, when shifting qubits
1103 * for density matrices)
1104 */
1105
1106 // global index of the first basis state in this node
1107 long long int firstInd = qureg.chunkId * qureg.numAmpsPerChunk;
1108
1109 /* optimisation: if this node doesn't contain any amplitudes for which {ctrls}={1}
1110 * (and hence, neither does the pair node), then these pair nodes have nothing to modify
1111 * nor any need to communicate, and can halt. No ctrls are contained in the node
1112 * if the distance from the first index, to the next index where {ctrls}=1, is
1113 * greater than the total contained amplitudes. This is a worthwhile optimisation,
1114 * since although we must still wait for the slowest node, we have potentially reduced
1115 * the network traffic and might avoid saturation.
1116 */
1117 if ((firstInd|ctrlMask) - firstInd >= qureg.numAmpsPerChunk)
1118 return;
1119
1120 /* nodes communicate pairwise, and (ignoring ctrls) swap all their amplitudes with mate.
1121 * hence we find |pairState> = X_{targs}|firstStateInNode>, determine which node contains
1122 * |pairState>, and swap state-vector with it (unless it happens to be this node).
1123 */
1124
1125 // global index of the corresponding NOT'd first basis state
1126 long long int pairInd = firstInd ^ targMask;
1127 int pairRank = pairInd / qureg.numAmpsPerChunk;
1128 int useLocalDataOnly = (pairRank == qureg.chunkId);
1129
1130 if (useLocalDataOnly) {
1131 // swaps amplitudes locally, setting |a>=X|b>, and |b>=X|a>
1132 statevec_multiControlledMultiQubitNotLocal(qureg, ctrlMask, targMask);
1133 } else {
1134 // swaps amplitudes with pair node
1135 exchangeStateVectors(qureg, pairRank);
1136 // modifies only |a>=X|b> (pair node handles the converse)
1137 statevec_multiControlledMultiQubitNotDistributed(
1138 qureg, ctrlMask, targMask,
1139 qureg.pairStateVec, // in
1140 qureg.stateVec); // out
1141 }

1516void statevec_multiControlledMultiQubitUnitary(Qureg qureg, long long int ctrlMask, int* targs, int numTargs, ComplexMatrixN u) {
1517
1518 // bit mask of target qubits (for quick collision checking)
1519 long long int targMask = getQubitBitMask(targs, numTargs);
1520
1521 // find lowest qubit available for swapping (isn't in targs)
1522 int freeQb=0;
1523 while (maskContainsBit(targMask, freeQb))
1524 freeQb++;
1525
1526 // assign indices of where each target will be swapped to (else itself)
1527 int swapTargs[numTargs];
1528 for (int t=0; t<numTargs; t++) {
1529 if (halfMatrixBlockFitsInChunk(qureg.numAmpsPerChunk, targs[t]))
1530 swapTargs[t] = targs[t];
1531 else {
1532 // mark swap
1533 swapTargs[t] = freeQb;
1534
1535 // update ctrlMask if swapped-out qubit was a control
1536 if (maskContainsBit(ctrlMask, swapTargs[t]))
1537 ctrlMask = flipBit(flipBit(ctrlMask, swapTargs[t]), targs[t]); // swap targ and ctrl
1538
1539 // locate next available on-chunk qubit
1540 freeQb++;
1541 while (maskContainsBit(targMask, freeQb))
1542 freeQb++;
1543 }
1544 }

1547 for (int t=0; t<numTargs; t++)
1548 if (swapTargs[t] != targs[t])
1549 statevec_swapQubitAmps(qureg, targs[t], swapTargs[t]);
1550
1551 // all target qubits have now been swapped into local memory
1552 statevec_multiControlledMultiQubitUnitaryLocal(qureg, ctrlMask, swapTargs, numTargs, u);
1553
1554 // undo swaps
1555 for (int t=0; t<numTargs; t++)

1561void copyDiagOpIntoMatrixPairState(Qureg qureg, DiagonalOp op) {
1562
1563 /* since, for every elem in 2^N op, there is a column in 2^N x 2^N qureg,
1564 * we know immediately (by each node containing at least 1 element of op)
1565 * that every node contains at least 1 column. Hence, we know that pairStateVec
1566 * of qureg can fit the entirety of op.
1567 */
1568
1569 // load up our local contribution
1570 long long int localOffset = qureg.chunkId * op.numElemsPerChunk;
1571 memcpy(&qureg.pairStateVec.real[localOffset], op.real, op.numElemsPerChunk * sizeof(qreal));
1572 memcpy(&qureg.pairStateVec.imag[localOffset], op.imag, op.numElemsPerChunk * sizeof(qreal));
1573
1574 // work out how many messages are needed to send op chunks (2GB limit)
1575 long long int maxMsgSize = MPI_MAX_AMPS_IN_MSG;
1576 if (op.numElemsPerChunk < maxMsgSize)
1577 maxMsgSize = op.numElemsPerChunk;
1578 int numMsgs = op.numElemsPerChunk / maxMsgSize; // since MPI_MAX... = 2^n, division is exact
1579
1580 // each node has a turn at broadcasting its contribution of op
1581 for (int broadcaster=0; broadcaster < qureg.numChunks; broadcaster++) {
1582 long long int broadOffset = broadcaster * op.numElemsPerChunk;
1583
1584 // (while keeping each message smaller than MPI max)
1585 for (int i=0; i<numMsgs; i++) {
1586 MPI_Bcast(
1587 &qureg.pairStateVec.real[broadOffset + i*maxMsgSize],
1588 maxMsgSize, MPI_QuEST_REAL, broadcaster, MPI_COMM_WORLD);
1589 MPI_Bcast(
1590 &qureg.pairStateVec.imag[broadOffset + i*maxMsgSize],
1591 maxMsgSize, MPI_QuEST_REAL, broadcaster, MPI_COMM_WORLD);
1592 }
1593 }
1594}
