56void qasm_setup(Qureg* qureg) {
57
58 // populate and attach QASM logger
59 QASMLogger *qasmLog = malloc(sizeof *qasmLog);
60 qureg->qasmLog = qasmLog;
61 if (qasmLog == NULL)
62 raiseQASMBufferOverflow(__func__);
63
64 qasmLog->isLogging = 0;
65 qasmLog->bufferSize = BUF_INIT_SIZE;
66 qasmLog->buffer = malloc(qasmLog->bufferSize * sizeof *(qasmLog->buffer));
67 if (qasmLog->buffer == NULL)
68 raiseQASMBufferOverflow(__func__);
69
70 // add headers and quantum / classical register creation
71 qasmLog->bufferFill = snprintf(
72 qasmLog->buffer, qasmLog->bufferSize,
73 "OPENQASM 2.0;\nqreg %s[%d];\ncreg %s[%d];\n",
74 QUREG_LABEL, qureg->numQubitsRepresented,
75 MESREG_LABEL, qureg->numQubitsRepresented);
76 if (qasmLog->bufferFill >= qasmLog->bufferSize)
77 raiseQASMBufferOverflow(__func__);
78}

88void addStringToQASM(Qureg qureg, char line[], int lineLen) {
89
90 char* buf = qureg.qasmLog->buffer;
91 int bufSize = qureg.qasmLog->bufferSize;
92 int bufFill = qureg.qasmLog->bufferFill;
93
94 // grow QASM buffer if necessary
95 if (lineLen + bufFill > bufSize) {
96
97 int newBufSize = BUF_GROW_FAC * bufSize;
98 if (lineLen + bufFill > newBufSize)
99 raiseQASMBufferOverflow(__func__);
100
101 char* newBuffer = malloc(newBufSize * sizeof *newBuffer);
102 sprintf(newBuffer, "%s", buf);
103 free(buf);
104
105 qureg.qasmLog->bufferSize = newBufSize;
106 qureg.qasmLog->buffer = newBuffer;
107 bufSize = newBufSize;
108 buf = newBuffer;
109 }
110
111 // add new str
112 int addedChars = snprintf(buf+bufFill, bufSize-bufFill, "%s", line);
113 qureg.qasmLog->bufferFill += addedChars;
114}

116void qasm_recordComment(Qureg qureg, char* comment, ...) {
117
118 if (!qureg.qasmLog->isLogging)
119 return;
120
121 // write formatted comment to buff
122 va_list argp;
123 va_start(argp, comment);
124 char buff[MAX_LINE_LEN - 4];
125 vsnprintf(buff, MAX_LINE_LEN-5, comment, argp);
126 va_end(argp);
127
128 // add chars to buff, write to QASM logger
129 char line[MAX_LINE_LEN + 1]; // for trailing \0
130 int len = snprintf(line, MAX_LINE_LEN, "%s %s\n", COMMENT_PREF, buff);
131 addStringToQASM(qureg, line, len);
132}

134void addGateToQASM(Qureg qureg, TargetGate gate, int* controlQubits, int numControlQubits, int targetQubit, qreal* params, int numParams) {
135
136 int len = 0;
137 char line[MAX_LINE_LEN + 1]; // for trailing \0
138
139 // add control labels
140 for (int i=0; i < numControlQubits; i++)
141 len += snprintf(line+len, MAX_LINE_LEN-len, "%s", CTRL_LABEL_PREF);
142
143 // add target gate
144 len += snprintf(line+len, MAX_LINE_LEN-len, "%s", qasmGateLabels[gate]);
145
146 // add parameters
147 if (numParams > 0) {
148 len += snprintf(line+len, MAX_LINE_LEN-len, "(");
149 for (int i=0; i < numParams; i++) {
150 len += snprintf(line+len, MAX_LINE_LEN-len, REAL_QASM_FORMAT, params[i]);
151 if (i != numParams - 1)
152 len += snprintf(line+len, MAX_LINE_LEN-len, ",");
153 }
154 len += snprintf(line+len, MAX_LINE_LEN-len, ")");
155 }
156
157 // add space
158 len += snprintf(line+len, MAX_LINE_LEN-len, " ");
159
160 // add control qubits
161 for (int i=0; i < numControlQubits; i++)
162 len += snprintf(line+len, MAX_LINE_LEN-len, "%s[%d],", QUREG_LABEL, controlQubits[i]);
163
164 // add target qubit, colon and newline
165 len += snprintf(line+len, MAX_LINE_LEN-len, "%s[%d];\n", QUREG_LABEL, targetQubit);
166
167 // check whether we overflowed buffer
168 if (len >= MAX_LINE_LEN)
169 raiseQASMBufferOverflow(__func__);
170
171 addStringToQASM(qureg, line, len);
172}

358void qasm_recordMultiStateControlledUnitary(
359 Qureg qureg, ComplexMatrix2 u, int* controlQubits, int* controlState, int numControlQubits, int targetQubit
360) {
361 if (!qureg.qasmLog->isLogging)
362 return;
363
364 qasm_recordComment(qureg, "NOTing some gates so that the subsequent unitary is controlled-on-0");
365 for (int i=0; i < numControlQubits; i++)
366 if (controlState[i] == 0)
367 addGateToQASM(qureg, GATE_SIGMA_X, NULL, 0, controlQubits[i], NULL, 0);
368
369 qasm_recordMultiControlledUnitary(qureg, u, controlQubits, numControlQubits, targetQubit);
370
371 qasm_recordComment(qureg, "Undoing the NOTing of the controlled-on-0 qubits of the previous unitary");
372 for (int i=0; i < numControlQubits; i++)
373 if (controlState[i] == 0)
374 addGateToQASM(qureg, GATE_SIGMA_X, NULL, 0, controlQubits[i], NULL, 0);
375}

485void qasm_recordPhaseFunc(Qureg qureg, int* qubits, int numQubits, enum bitEncoding encoding, qreal* coeffs, qreal* exponents, int numTerms, long long int* overrideInds, qreal* overridePhases, int numOverrides) {
486
487 if (!qureg.qasmLog->isLogging)
488 return;
489
490 qasm_recordComment(qureg, "Here, applyPhaseFunc() multiplied a complex scalar of the form");
491
492 // record like:
493 // exp(i (-.5 x^2 + .5 x^(-1.5) - 1.3 x^4 ))
494 char line[MAX_LINE_LEN+1];
495 int len = snprintf(line, MAX_LINE_LEN, "// exp(i (");
496 for (int t=0; t<numTerms; t++) {
497 len += snprintf(line+len, MAX_LINE_LEN-len,
498 (exponents[t] > 0)?
499 (REAL_QASM_FORMAT " x^" REAL_QASM_FORMAT) :
500 (REAL_QASM_FORMAT " x^(" REAL_QASM_FORMAT ")"),
501 (t>0)?
502 absReal(coeffs[t]) :
503 coeffs[t],
504 exponents[t]);
505 if (t < numTerms-1)
506 len += snprintf(line+len, MAX_LINE_LEN-len, (coeffs[t+1] > 0)? " + ":" - ");
507 }
508 len += snprintf(line+len, MAX_LINE_LEN-len, "))\n");
509
510 if (len >= MAX_LINE_LEN)
511 raiseQASMBufferOverflow(__func__);
512 addStringToQASM(qureg, line, len);
513
514 char encBuf[MAX_LINE_LEN];
515 if (encoding == UNSIGNED) sprintf(encBuf, "an unsigned");
516 if (encoding == TWOS_COMPLEMENT) sprintf(encBuf, "a two's complement");
517 qasm_recordComment(qureg, " upon every substate |x>, informed by qubits (under %s binary encoding)", encBuf);
518
519 // record like:
520 // {0, 3, 2}
521 len=0;
522 len = snprintf(line, MAX_LINE_LEN, "// {");
523 for (int q=0; q<numQubits; q++)
524 len += snprintf(line+len, MAX_LINE_LEN-len, (q < numQubits-1)? "%d, ":"%d}\n", qubits[q]);
525
526 if (len >= MAX_LINE_LEN)
527 raiseQASMBufferOverflow(__func__);
528 addStringToQASM(qureg, line, len);
529
530 if (numOverrides > 0) {
531 // optionally record like:
532 // |0> -> exp(i .45)
533 // |1> -> exp(i (-.5))
534 qasm_recordComment(qureg, " though with overrides");
535 for (int v=0; v<numOverrides; v++)
536 qasm_recordComment(qureg, (overridePhases[v] >= 0)?
537 " |%lld> -> exp(i " REAL_QASM_FORMAT ")" :
538 " |%lld> -> exp(i (" REAL_QASM_FORMAT "))",
539 overrideInds[v], overridePhases[v]);
540 }
541
542 // Here, applyPhaseFunction() multiplied a complex scalar of the form
543 // exp(i (.5 x^2 + .5 x^(-1.5) - 1.3 x^4 ))
544 // upon every sub-state |x>, informed by qubits
545 // {0, 1, 2}
546 // though with overrides
547 // |0> -> exp(i .45)
548 // |1> -> exp(i (-.5))
549}

551char getPhaseFuncSymbol(int numSymbs, int ind) {
552
553 static char xyz[7] = {'x', 'y', 'z', 't', 'r', 'v', 'u'};
554 if (numSymbs <= 7)
555 return xyz[ind];
556
557 static char abc[MAX_REG_SYMBS] = {'a','b','c','d','e','f','g','h','j','k','l','m','n','p','q','r','s','t','u','v','w','x','y','z'}; // no i or o
558 if (numSymbs <= MAX_REG_SYMBS)
559 return abc[ind];
560
561 // we should never reach here, since caller should handle when numSymbs > 24
562 raiseQASMBufferOverflow(__func__);
563 return 'x';
564}

566void addMultiVarRegsToQASM(Qureg qureg, int* qubits, int* numQubitsPerReg, int numRegs, enum bitEncoding encoding) {
567
568 char encBuf[MAX_LINE_LEN];
569 if (encoding == UNSIGNED) sprintf(encBuf, "an unsigned");
570 if (encoding == TWOS_COMPLEMENT) sprintf(encBuf, "a two's complement");
571 qasm_recordComment(qureg, " upon substates informed by qubits (under %s binary encoding)", encBuf);
572
573 char line[MAX_LINE_LEN+1];
574 int len = 0;
575
576 // record like:
577 // |x> = {0, 3, 2}
578 // |y> = {1, 2}
579 int qInd = 0;
580 for (int r=0; r<numRegs; r++) {
581 len = 0;
582 if (numRegs <= MAX_REG_SYMBS)
583 len += snprintf(line+len, MAX_LINE_LEN-len, "// |%c> = {", getPhaseFuncSymbol(numRegs,r));
584 else
585 len += snprintf(line+len, MAX_LINE_LEN-len, "// |x%d> = {", r);
586 for (int q=0; q<numQubitsPerReg[r]; q++)
587 len += snprintf(line+len, MAX_LINE_LEN-len, (q < numQubitsPerReg[r]-1)? "%d, ":"%d}\n", qubits[qInd++]);
588
589 if (len >= MAX_LINE_LEN)
590 raiseQASMBufferOverflow(__func__);
591 addStringToQASM(qureg, line, len);
592 }
