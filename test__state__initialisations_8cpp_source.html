189 QMatrix matRef = getZeroMatrix(1<<NUM_QUBITS);
190 for (size_t i=0; i<matRef.size(); i++)
191 for (size_t j=0; j<matRef.size(); j++)

451TEST_CASE( "setDensityAmps", "[state_initialisations]" ) {
452
453 Qureg matr = createDensityQureg(NUM_QUBITS, QUEST_ENV);
454 VLA(qreal, reals, matr.numAmpsTotal);
455 VLA(qreal, imags, matr.numAmpsTotal);
456 int maxInd = (1 << NUM_QUBITS);
457
458 SECTION( "correctness" ) {
459
460 SECTION( "density-matrix" ) {
461
462 // try all valid number of amplitudes and offsets
463 int startRow = GENERATE_COPY( range(0,maxInd) );
464 int startCol = GENERATE_COPY( range(0,maxInd) );
465
466 // determine the max number of amps that can be passed from the given start indices
467 int numPriorAmps = startRow + startCol*(1 << matr.numQubitsRepresented);
468 int maxNumAmps = matr.numAmpsTotal - numPriorAmps;
469
470 // previously, we tried all possible number of amps, like so:
471 // int numAmps = GENERATE_COPY( range(0,maxNumAmps) ); // upper-bound allows all amps specified
472 // but this is too many and causes a timeout on Ubuntu.
473 // So we instead randomly choose the number of amps, and repeat 10 times.
474 int numAmps = getRandomInt(0, maxNumAmps);
475 GENERATE_COPY( range(0,10) );
476
477 // generate random amplitudes
478 for (int i=0; i<numAmps; i++) {
479 reals[i] = getRandomReal(-5,5);
480 imags[i] = getRandomReal(-5,5);
481 }
482
483 // check both specified and un-specified amplitudes are correctly handled
484 initDebugState(matr);
485 QMatrix matrRef = toQMatrix(matr);
486
487 setDensityAmps(matr, startRow, startCol, reals, imags, numAmps);
488
489 int r=startRow;
490 int c=startCol;
491 for (int i=0; i<numAmps; i++) {
492 qcomp amp = reals[i] + imags[i] * (qcomp) 1i;
493 matrRef[r][c] = amp;
494
495 r++;
496 if (r >= maxInd ) {
497 r=0;
498 c++;
499 }
500 }
501
502 REQUIRE( areEqual(matr, matrRef) );
503 }
504 }

601
602 SECTION( "correctness" ) {
603
604 // repeat each test below 10 times
605 GENERATE( range(0,10) );
606
607 /* note tolerance in areEqual increases with tests, since
608 * small differences propogate in vecC which is not re-initialised
609 */
610
611 SECTION( "state-vector" ) {
612
613 // make three random vectors
614 Qureg vecA = createQureg(NUM_QUBITS, QUEST_ENV);
615 Qureg vecB = createQureg(NUM_QUBITS, QUEST_ENV);
616 Qureg vecC = createQureg(NUM_QUBITS, QUEST_ENV);
617 for (int j=0; j<vecA.numAmpsPerChunk; j++) {
618 vecA.stateVec.real[j] = getRandomReal(-5,5); vecA.stateVec.imag[j] = getRandomReal(-5,5);
619 vecB.stateVec.real[j] = getRandomReal(-5,5); vecB.stateVec.imag[j] = getRandomReal(-5,5);
620 vecC.stateVec.real[j] = getRandomReal(-5,5); vecC.stateVec.imag[j] = getRandomReal(-5,5);
621 }
622 copyStateToGPU(vecA); copyStateToGPU(vecB); copyStateToGPU(vecC);
623 QVector refA = toQVector(vecA);
624 QVector refB = toQVector(vecB);
625 QVector refC = toQVector(vecC);
626 QVector refOut;
627
628 // get three random factors
629 qcomp numA = getRandomReal(-5,5) + getRandomReal(-5,5) * (qcomp) 1i;
630 qcomp numB = getRandomReal(-5,5) + getRandomReal(-5,5) * (qcomp) 1i;
631 qcomp numC = getRandomReal(-5,5) + getRandomReal(-5,5) * (qcomp) 1i;
632 Complex facA; facA.real = real(numA); facA.imag = imag(numA);
633 Complex facB; facB.real = real(numB); facB.imag = imag(numB);
634 Complex facC; facC.real = real(numC); facC.imag = imag(numC);
635
636 // check out-qureg is correct, when all quregs are unique...
637 setWeightedQureg(facA, vecA, facB, vecB, facC, vecC);
638 refOut = numA*refA + numB*refB + numC*refC;
639 REQUIRE( areEqual(vecC, refOut) );
640
641 // ... and that other qureg's aren't modified
642 REQUIRE( areEqual(vecA, refA) );
643 REQUIRE( areEqual(vecB, refB) );
644
645 // check quregOut correct, when it's also qureg2
646 refC = toQVector(vecC);
647 setWeightedQureg(facB, vecB, facC, vecC, facA, vecC);
648 refOut = numB*refB + numC*refC + numA*refC;
649 REQUIRE( areEqual(vecC, refOut, 10*REAL_EPS) );
650
651 // ... and that the remaining qureg is not modified
652 REQUIRE( areEqual(vecB, refB) );
653
654 // check quregOut correct, when it's also qureg1
655 refC = toQVector(vecC);
656 setWeightedQureg(facC, vecC, facB, vecB, facA, vecC);
657 refOut = numC*refC + numB*refB + numA*refC;
658 REQUIRE( areEqual(vecC, refOut, 10*REAL_EPS) );
659
660 // ... and that the remaining qureg is not modified
661 REQUIRE( areEqual(vecB, refB) );
662
663 // check quregOut is correct when it's both input quregs
664 refC = toQVector(vecC);
665 setWeightedQureg(facA, vecC, facB, vecC, facC, vecC);
666 refOut = numA*refC + numB*refC + numC*refC;
667 REQUIRE( areEqual(vecC, refOut, 1E3*REAL_EPS) );
668
669 // cleanup
670 destroyQureg(vecA, QUEST_ENV);
671 destroyQureg(vecB, QUEST_ENV);
672 destroyQureg(vecC, QUEST_ENV);
673 }

678 Qureg matB = createDensityQureg(NUM_QUBITS, QUEST_ENV);
679 Qureg matC = createDensityQureg(NUM_QUBITS, QUEST_ENV);
680 for (int j=0; j<matA.numAmpsPerChunk; j++) {

688 QMatrix refC = toQMatrix(matC);

692 qcomp numA = getRandomReal(-5,5) + getRandomReal(-5,5) * (qcomp) 1i;
693 qcomp numB = getRandomReal(-5,5) + getRandomReal(-5,5) * (qcomp) 1i;

695 Complex facA; facA.real = real(numA); facA.imag = imag(numA);

716
